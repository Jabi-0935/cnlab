#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <stdlib.h>

int i, j, n;
int x[15], y[15], z[15];
float distance[15][15];

void createNetwork() {
    int x_range, y_range, z_range;

    printf("Enter the number of nodes (max 15): ");
    scanf("%d", &n);

    printf("Enter X-Coordinate range: ");
    scanf("%d", &x_range);

    printf("Enter Y-Coordinate range: ");
    scanf("%d", &y_range);
    printf("Enter Z-Coordinate range: ");
    scanf("%d", &z_range);
    

    srand(time(NULL));

    for (i = 0; i < n; i++) {
        x[i] = 1 + rand() % x_range;
        y[i] = 1 + rand() % y_range;
        z[i] = 1 + rand() % z_range;
        
    }

    printf("\nX\tY\tZ\n");
    for (i = 0; i < n; i++) {
        printf("(%d,\t%d,\t%d)\n", x[i], y[i],z[i]);
    }
}

void computeDistance() {
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            distance[i][j] = sqrt(pow(x[j] - x[i], 2) + pow(y[j] - y[i], 2)+pow(z[j] - z[i], 2));
        }
    }

    printf("\t");
    for (i = 0; i < n; i++) {
        printf("%d\t", i + 1);
    }
    printf("\n");

    for (i = 0; i < n; i++) {
        printf("%d\t", i + 1);
        for (j = 0; j < n; j++) {
            printf("%.2f\t", distance[i][j]);
        }
        printf("\n");
    }
}

void findNeighbor() {
    float range;

    printf("\nEnter the neighbor range: ");
    scanf("%f", &range);

    for (i = 0; i < n; i++) {
        printf("\nNeighbors of node %d are:\t", i + 1);
        for (j = 0; j < n; j++) {
            if (i != j && distance[i][j] <= range) {
                printf("%d (distance: %.2f)\t", j + 1, distance[i][j]);
            }
        }
        printf("\n");
    }
}

int main(int argc, char* argv[]) {
    createNetwork();
    computeDistance();
    findNeighbor();
    return EXIT_SUCCESS;
}

#include <stdio.h>
#include <stdlib.h>

int distance[20], parent[20];

int findShortestPath(int G[6][6], int Vertices, int Edges, int edge[20][2], int Source) {
    for (int i = 0; i < Vertices; i++) {
        distance[i] = 999;
        parent[i] = -1;
    }
    distance[Source - 1] = 0;

    for (int i = 0; i < Vertices - 1; i++) {
        for (int k = 0; k < Edges; k++) {
            int u = edge[k][0];
            int v = edge[k][1];
            if (distance[u] + G[u][v] < distance[v]) {
                distance[v] = distance[u] + G[u][v];
                parent[v] = u;
            }
        }
    }

    int flag = 1;
    for (int k = 0; k < Edges; k++) {
        int u = edge[k][0];
        int v = edge[k][1];
        if (distance[u] + G[u][v] < distance[v]) {
            flag = 0;
            break;
        }
    }

    printf("Destination\tCost\tPath\n");
    for (int i = 0; i < Vertices; i++) {
        if (distance[i] != 999) {
            printf("%d\t\t%d\t", i, distance[i]);
            int nodes[10], b = 9, next = parent[i];
            nodes[b--] = i;
            while (next != -1) {
                nodes[b--] = next;
                next = parent[next];
            }
            for (int h = b + 1; h < 10; h++) {
                if (nodes[h] != -1) {
                    if (h == b + 1)
                        printf("%d", nodes[h]);
                    else
                        printf(" --> %d", nodes[h]);
                }
            }
            printf("\n");
        }
    }

    return flag;
}

                edge[Edges][0] = i;
                edge[Edges][1] = j;
                Edges++;
            }
        }
        printf("\n");
    }

    for (int i = 0; i < Vertices; i++) {
        printf("\nSource vertex %d:\n", i + 1);
        findShortestPath(graph, Vertices, Edges, edge, i + 1);
    }

}

#include <stdio.h>
#include <stdlib.h>

// Function declarations
void StopAndWait();
void GoBackN();
void Selective();

int main(int argc, char** argv) {
    int val;
    while (1) {
        printf("\nEnter 1.Stop And Wait protocol\n"
                "2.Go back N\n3.Selective repeat\n");
        scanf("%d", &val);
        switch (val) {
            case 1:
                StopAndWait();
                break;
            case 2:
                GoBackN();
                break;
            case 3:
                Selective();
                break;
            default:
                return 0;
        }
    }
    return (EXIT_SUCCESS);
}

void StopAndWait() {
    int n, i = 0, frame;
    printf("Enter the total number of frames ");
    scanf("%d", &n);
    while (i != n) {
        printf("\nEnter received frame ");
        scanf("%d", &frame);
        if (frame == i + 1) {
            printf("Transmitting..... ACK to frame %d\n", frame);
            i++;
        } else {
            printf("Negative ACK.... to frame %d\n", i + 1);
        }
    }
}

void GoBackN() {
    int n, i = 0, frame, size, t;
    printf("Enter the total number of frames ");
    scanf("%d", &n);
    printf("Enter window size of frames ");
    scanf("%d", &size);
    printf("Sending frames ");
    for (int j = 0; j < size; j++) {
        printf("%d,", j + 1);
    }
    for (int j = 0; j < n; j++) {
        printf("\nIs Frame %d received (1 or 0)? ", j + 1);
        scanf("%d", &t);
        if (t == 1) {
            printf("\nSending ACK to frame %d\nsliding window", j + 1);
            printf("\nIn window ");
            for (int k = j + 1; k < j + 1 + size; k++) {
                if (k < n)
                    printf("%d ", k + 1);
            }
        } else {
            printf("\nRetransmitting frames\t");
            for (int m = j; m < j + size; m++) {
                if (m < n)
                    printf("%d ", m + 1);
            }
            j--;
        }
    }
}

void Selective() {
    int n, i = 0, frame, m, size;
    printf("Enter the total number of frames ");
    scanf("%d", &n);
    printf("Sending frames ");
    for (int j = 0; j < n; j++) {
        printf("%d,", j + 1);
    }
    printf("\nEnter the number of frames received\n");
    scanf("%d", &m);
    int a[n];
    printf("\nEnter received frame\n");
    for (int i = 0; i < m; i++) {
        scanf("%d", &a[i]);
        int trigger = 0;
        for (int i = 0; i < n; i++) {
            trigger = 0;
            for (int j = 0; j < m; j++) {
                if ((i + 1) == a[j]) {
                    trigger = 1;
                }
            }
            if (trigger == 0) {
                printf("Retransmitting frame %d\n", i + 1);
                a[m++] = i + 1;
            }
        }
    }

    // Sorting frames
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (a[i] < a[j]) {
                int temp = a[i];
                a[i] = a[j];
                a[j] = temp;
            }
        }
    }

    printf("\nSorting ");
    for (int i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
}
